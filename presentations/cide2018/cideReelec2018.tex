\section{El cambio electoral}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame {                      % SLIDE
  \frametitle{La reforma de 2014}
  Sorprendió la remoción de la no reelección consecutiva
  \bigskip
  \begin{itemize}
  \item Dip. federales hasta 4 periodos consecutivos
  \item Senadores hasta 2 periodos
  \item Ocupantes deben ser renominados por el partido que los eligió
  \item Patada inicial: intermedia de 2021
  \end{itemize}
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame {                      % SLIDE
  \frametitle{El nivel subnacional}

  Reformadores dieron discreción institucional a los estados
  \bigskip
  \begin{itemize}
  \item Diputados locales: 2, 3, o 4 periodos consecutivos
  \item Alcaldes y regidores: 1 o 2 periodos consecutivos
  \item Candado partidista obligatorio
  \item Inaplicable para los propios reformadores
  \end{itemize}
  \bigskip %\pause
  Varianza de los calendarios electorales $\rightarrow$ ocupantes accederán a la boleta \alert{progresivamente} 
}

\frame {                      % SLIDE

    \frametitle{Urgency authority in Chilean constitution and law}

\bi
\item Any bill at any stage can be declared urgent

\item Chamber must ``discuss and vote'' before 30 days

\item Law sets the breadth of the interference

\be
\item `act now'           (\emph{discusión inmediata}, 6 days)
\item `two weeks' notice  (\emph{urgencia suma}, 15 days)
\item `four weeks' notice (\emph{urgencia simple}, 30 days) 
\ee

\item Can retire the urgency, with immediate effects

\item Non-compliance: reversionary schedule/policy indeterminate
  \ei
